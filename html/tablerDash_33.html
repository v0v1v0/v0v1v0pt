tablerTimeline R Documentation

Create a Boostrap 4 timeline

Description

Build a tabler timeline

Usage

tablerTimeline(...)

Arguments

...

slot for tablerTimelineItem.

Author(s)

David Granjon, dgranjon@ymail.com

Examples

if(interactive()){
 library(shiny)
 library(tablerDash)

 shiny::shinyApp(
   ui = tablerDashPage(
    navbar = NULL,
    footer = NULL,
    title = "test",
    body = tablerDashBody(
     tablerTimeline(
      tablerTimelineItem(
       title = "Item 1",
       status = "green",
       date = "now"
      ),
      tablerTimelineItem(
       title = "Item 2",
       status = NULL,
       date = "yesterday",
       "Lorem ipsum dolor sit amet,
       consectetur adipisicing elit."
      )
     )
    )
   ),
   server = function(input, output) {}
 )
}
