## S4 method for signature 'H2OBinomialModel'

  standardized = FALSE,
  include.mse = TRUE,

  include.meanerror = TRUE,
  include.auc = TRUE,
  include.gini = TRUE,
  include.deviance = TRUE,
  include.aic = TRUE,
  ...
)
